```typescript title="src/mastra/agents/my-agent.ts"
  id: "custom-agent",
  name: "custom-agent",
  model: {
    url: "https://api.clarifai.com/v2/ext/openai/v1",
    id: "clarifai/arcee_ai/AFM/models/trinity-mini",
    apiKey: process.env.CLARIFAI_PAT,

      "X-Custom-Header": "value"
    }
  }
});
```

```typescript title="src/mastra/agents/my-agent.ts"
});
```
