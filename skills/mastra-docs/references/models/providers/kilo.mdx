### Dynamic model selection

```typescript title="src/mastra/agents/my-agent.ts"
const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "kilo/z-ai/glm-5"
      : "kilo/ai21/jamba-large-1.7";
  }
});
```
