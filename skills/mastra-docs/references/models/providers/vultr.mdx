const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "vultr/gpt-oss-120b"
      : "vultr/DeepSeek-R1-Distill-Llama-70B";
  }
});
