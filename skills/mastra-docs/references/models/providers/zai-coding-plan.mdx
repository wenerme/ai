const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "zai-coding-plan/glm-5"
      : "zai-coding-plan/glm-4.5";
  }
});
